## 1 Создание Документа

```js
&НаКлиенте
Процедура СоздатьДокумент(Команда)
	СоздатьДокументНаСервере();
	Сообщить("Документ успешно создан");
КонецПроцедуры

&НаСервере
Процедура СоздатьДокументНаСервере()
	
	НовыйДокумент = Документы.ПоступлениеТоваров.СоздатьДокумент();
	НовыйДокумент.Дата = ТекущаяДата();
	НовыйДокумент.Контрагент = Справочники.Контрагенты.НайтиПоНаименованию("ООО КосмоТефтелина");
	НовыйДокумент.СуммаДокумента = 5000;
	
	СрокаТовары = НовыйДокумент.Товары.Добавить();
	СрокаТовары.Номенклатура = Справочники.Номенклатура.НайтиПоНаименованию("Хлорики");
	СрокаТовары.Количество = 5;
	СрокаТовары.Цена = 40;
	СрокаТовары.Сумма = СрокаТовары.Количество * СрокаТовары.Цена;
	
	НовыйДокумент.Записать(РежимЗаписиДокумента.Проведение);
		
КонецПроцедуры
```

## 2 Поиск Документа

```js
&НаКлиенте
Процедура НайтиДокумент(Команда)
	НайтиДокументНаСервере();
КонецПроцедуры


&НаСервере
Процедура НайтиДокументНаСервере()
	
	Контрагент = Справочники.Контрагенты.НайтиПоНаименованию("ООО КосмоТефтелина");
	
	Если Контрагент = Справочники.Контрагенты.ПустаяСсылка() Тогда
		Сообщить("Контрагент не найден");
		Возврат;	
	КонецЕсли;
	
	НайденныйДокумент = Документы.ПоступлениеТоваров.НайтиПоРеквизиту("Контрагент", Контрагент);
	
	Если НайденныйДокумент = Документы.ПоступлениеТоваров.ПустаяСсылка() Тогда
		Сообщить("Документ не найден");
		Возврат;	
	КонецЕсли;
	
	Сообщить(НайденныйДокумент.Дата);
	
КонецПроцедуры
```

## 3 Изменение Документа

```js
&НаКлиенте
Процедура ИзменитьДокумент(Команда)
	ИзменитьДокументНаСервере();
	Сообщить("Документ успешно изменён");
КонецПроцедуры


&НаСервере
Процедура ИзменитьДокументНаСервере()
	НайденныйДокумент = Документы.ПоступлениеТоваров.НайтиПоНомеру("000000001");
	
	Если НайденныйДокумент = Документы.ПоступлениеТоваров.ПустаяСсылка() Тогда
		Сообщить("Документ не найден");
		Возврат;	
	КонецЕсли;
	
	ОбъектДокумента = НайденныйДокумент.ПолучитьОбъект();
	ОбъектДокумента.СуммаДокумента = 300;
	
	ОбъектДокумента.Записать(РежимЗаписиДокумента.Проведение);
	
КонецПроцедуры
```
