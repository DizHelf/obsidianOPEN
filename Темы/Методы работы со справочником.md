## 1 Создание элемента справочника

```js
&НаКлиенте
Процедура СоздатьЭлемент(Команда)
	
	СоздатьЭлементНаСервере();
	
КонецПроцедуры

&НаСервере
Процедура СоздатьЭлементНаСервере()
	
	//создание нового элемента Номенклатура
	Номенклатура = Справочники.Номенклатура.СоздатьЭлемент();
	
	//Ввод данных
	Номенклатура.Наименование = "Ножь";
	Номенклатура.Артикул = "12312";
	
	//Запись
	Номенклатура.Записать();
	
КонецПроцедуры
```
